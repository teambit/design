import { ButtonLink } from './button-link';

A component that renders a [link](https://bit.cloud/teambit/base-react/navigation/link) with a button design.

### Component usage

```js live
<ButtonLink
  linkProps={{ external: true, href: 'https://bit.cloud/teambit' }}
  icon={<img src="https://static.bit.cloud/cloud/organization.svg" />}
>
  Teambit
</ButtonLink>
```
