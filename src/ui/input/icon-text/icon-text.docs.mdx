import { useState } from 'react';
import { IconText } from './icon-text';

### Usages

IconText component based on Text component.  
Supports all props from native html input.  
Size support with em, the container of the component can use `font-size` to change the size of the component.

### Icon and Text

```js live
() => {
  const [value, setValue] = useState('');
  return <IconText placeholder="search..." icon="discovery" onChange={(e) => setValue(e.target.value)} />;
};
```
