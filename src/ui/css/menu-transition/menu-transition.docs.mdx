import classNames from 'classnames';
import { Dropdown } from '@teambit/design.inputs.dropdown';
import { MenuItem } from '@teambit/design.inputs.selectors.menu-item';
import exampleStyles from './menu-transition.example.module.scss';
import { menuTransition, rightMenuTransition, centerMenuTransition } from './';

## Overview:

This is a css only component. use it to add effects to any dropdown that toggles using the `data-open` attribute.

## Usage:

```js live
<Dropdown dropClass={classNames(exampleStyles.dropClass, menuTransition)} placeholder="left transition effect">
  <MenuItem>user</MenuItem>
  <MenuItem>user</MenuItem>
  <MenuItem>user</MenuItem>
</Dropdown>
```

```js live
<Dropdown dropClass={classNames(exampleStyles.dropClass, rightMenuTransition)} placeholder="right transition effect">
  <MenuItem>user</MenuItem>
  <MenuItem>user</MenuItem>
  <MenuItem>user</MenuItem>
</Dropdown>
```

```js live
<Dropdown dropClass={classNames(exampleStyles.dropClass, centerMenuTransition)} placeholder="center transition effect">
  <MenuItem>user</MenuItem>
  <MenuItem>user</MenuItem>
  <MenuItem>user</MenuItem>
</Dropdown>
```
